const { Api, TelegramClient, client } = require("telegram");
const { StringSession } = require("telegram/sessions");
const fs = require("fs").promises;
const path = require("path");
const mongoose = require("mongoose");
const input = require("./utils/input.js");
const dotenv = require("dotenv").config({ path: "configuration files/.env" });
const TARGET_CHANNEL = process.env.TARGET_CHANNEL;
const keywords = require("./utils/keywords.js");
const { sendToChannel } = require("./utils/sendToChannel.js");

(async () => {
  const SESSION_FILE = process.env.SESSION_FILE;
  const stringSession = new StringSession(process.env.SESSION);

  try {
    stringSession = await fs.readFile(SESSION_FILE, "utf8");
  } catch (e) {
    console.log("session not found (new login)");
  }

  const client = new TelegramClient(
    stringSession,
    parseInt(process.env.API_ID, 10),
    process.env.API_HASH,
    { connectionRetries: 5 }
  );

  // 🔌 Login
  console.log("🔗Connecting to Telegram...🔗");
  await client.start({
    phoneNumber: async () => process.env.PHONE_NUMBER,
    password: async () => "",
    phoneCode: async () => {
      console.log("\n📲 Check Telegram for the 5-digit login code.");
      return await input.text("Enter code: ");
    },
    onError: (err) => {
      console.error("💥 Login failed:", err.message);
      process.exit(1);
    },
  });

  // 💾 Save session for next time — AFTER successful login
  const newSession = client.session.save();
  if (typeof newSession === "string") {
    await fs.writeFile(SESSION_FILE, newSession);
    console.log(`✅ Session saved. Logged in as: ${client.user?.firstName}`);
  } else {
    console.error("❌ Failed to save session — session data is not a string.");
  }

  // 🔍 Search keywords (Persian + English — Telegram supports both)
  const results = [];

  for (const keyword of keywords) {
    console.log(`\n🔍 Searching for: "${keyword}"`);
    try {
      const res = await client.invoke(
        new Api.contacts.Search({ q: keyword, limit: 5 })
      );
      for (const chat of res.chats || []) {
        // Only public channels/groups with username
        if (chat.className !== "Channel" || !chat.username) continue;

        try {
          const full = await client.invoke(
            new Api.channels.GetFullChannel({ channel: chat })
          );
          const members = full.fullChat.participantsCount || 0;
          const lastPostTs = full.fullChat.participants;
          const daysAgo = lastPostTs
            ? Math.floor((Date.now() - lastPostTs * 1000) / 864e5)
            : Infinity;

          // Simple scoring logic
          let score = 0;
          if (members <= 20) score += 60; // size
          if (daysAgo > 30) score += 20; // inactivity
          if (full.fullChat.defaultSendAs) score += 10;
          if (full.fullChat.adminsCount > 1) score += 10;

          if (score >= 50) {
            results.push({
              title: chat.title,
              username: `@${chat.username}`,
              members: members,
              silentDays: daysAgo,
              score: score,
            });

            console.log(
              `✅ ${chat.title} | ${members} members | silent ${daysAgo} days | score: ${score}`
            );
          }

          await sendToChannel(client, TARGET_CHANNEL, full);
        } catch (e) {
          console.log(e.message);
        }

        async function sendToChannel(client, targetChannel, full) {
          if (!client || !targetChannel) {
            console.error("Client or Target channel is not set.");
            return;
          }

          try {
            // 2. حتماً از AWAIT استفاده کنید
            await client.sendMessage(targetChannel, {
              message: `Title: ${full.title} \nUsername: @${full.username}`,
            });
            console.log(
              `✅ A new group (${full.title}) sent to ${targetChannel}`
            );
          } catch (err) {
            console.error("❌ Cannot send message to channel:", err.message);
          }
        }

        await new Promise((r) => setTimeout(r, 1000)); // avoid flood
      }
    } catch (e) {
      console.warn(`⚠️ Skipped search for "${keyword}"`);
    }
    await new Promise((r) => setTimeout(r, 2000));
  }

  // 💾 Save results to JSON
  await fs.writeFile(
    "groups_for_sale.json",
    JSON.stringify(...results, null, 2),
    "utf8"
  );

  console.log(`\n🎉 Done! ${results.length} promising groups found.`);
  console.log("📁 Results saved to: groups_for_sale.json");

  await client.disconnect();
  console.log("👋 Session closed. Good luck with your purchases!");
})();
